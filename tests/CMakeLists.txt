find_package(GTest REQUIRED)

function(add_flox_test name)
  add_executable(${name} ${name}.cpp)
  target_include_directories(${name}
    PRIVATE
        include()
  )
  target_link_libraries(${name}
    PRIVATE ${FLOX}
    GTest::gtest
    GTest::gtest_main
    GTest::gmock
    GTest::gmock_main
    Threads::Threads
  )
  add_test(NAME ${name} COMMAND ${name})
endfunction()
